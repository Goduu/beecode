  type: Debugging
  description: "Identify the issue in the following `while` loop code snippet where the user is prompted for input, and suggest a correction."
  codeSnippet: |
    let numeroSecreto = 5;
    let chute = prompt('Escolha um número entre 1 e 100');

    while (chute != numeroSecreto) {
        chute = prompt('Escolha um número entre 1 e 100');
        // se chute for igual ao número secreto
        if (chute == numeroSecreto) { 
            alert(`Isso ai! Você descobriu o número secreto ${numeroSecreto}`);
        } else {
            if (chute > numeroSecreto) {
                alert(`O número secreto é menor que ${chute}`);
            } else {
                alert(`O número secreto é maior que ${chute}`);
            }
        }
    }
  expectedOutput: "The `while` loop should continue until the correct number is guessed. The initial prompt should be inside the loop."
  correctAnswer: |
    Issue: The initial prompt is outside the loop, causing it to only execute once before the loop starts.
    Correction: Move the prompt inside the loop so that the user is continuously prompted until they guess the correct number.
